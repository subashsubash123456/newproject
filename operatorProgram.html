About Java

Java is a high-level, object-oriented programming language originally developed by Sun Microsystems. It is known for its "write once, run anywhere" (WORA) philosophy, which means Java code can be executed on any platform that has a Java Virtual Machine (JVM). Java is widely used in web development, mobile app development, enterprise software, and more.

Program to display all operators
public class AllOperatorsExample {
    public static void main(String[] args) {
        // Arithmetic Operators
        int a = 10, b = 5;
        System.out.println("Arithmetic Operators:");
        System.out.println("a + b = " + (a + b));
        System.out.println("a - b = " + (a - b));
        System.out.println("a * b = " + (a * b));
        System.out.println("a / b = " + (a / b));
        System.out.println("a % b = " + (a % b));

        // Assignment Operators
        int c = 15;
        c += 3;
        System.out.println("\nAssignment Operators:");
        System.out.println("c += 3 (c = c + 3) => c = " + c);

        // Increment and Decrement Operators
        int d = 7;
        System.out.println("\nIncrement and Decrement Operators:");
        System.out.println("d++ (Post-increment) => d = " + d++);
        System.out.println("++d (Pre-increment) => d = " + ++d);
        System.out.println("d-- (Post-decrement) => d = " + d--);
        System.out.println("--d (Pre-decrement) => d = " + --d);

        // Relational Operators
        System.out.println("\nRelational Operators:");
        System.out.println("a == b: " + (a == b));
        System.out.println("a != b: " + (a != b));
        System.out.println("a < b: " + (a < b));
        System.out.println("a > b: " + (a > b));
        System.out.println("a <= b: " + (a <= b));
        System.out.println("a >= b: " + (a >= b));

        // Logical Operators
        boolean x = true, y = false;
        System.out.println("\nLogical Operators:");
        System.out.println("x && y: " + (x && y));
        System.out.println("x || y: " + (x || y));
        System.out.println("!x: " + !x);
        System.out.println("!y: " + !y);

        // Bitwise Operators
        int num1 = 5, num2 = 3;
        System.out.println("\nBitwise Operators:");
        System.out.println("num1 & num2: " + (num1 & num2));
        System.out.println("num1 | num2: " + (num1 | num2));
        System.out.println("num1 ^ num2: " + (num1 ^ num2));
        System.out.println("~num1: " + (~num1));
        System.out.println("num1 << 1: " + (num1 << 1));
        System.out.println("num1 >> 1: " + (num1 >> 1));
        System.out.println("num1 >>> 1: " + (num1 >>> 1));

        // Conditional Operator (Ternary Operator)
        int age = 25;
        String status = (age >= 18) ? "Adult" : "Minor";
        System.out.println("\nConditional Operator (Ternary Operator):");
        System.out.println("Status: " + status);
    }
}
